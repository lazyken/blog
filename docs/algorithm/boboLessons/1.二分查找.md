# 二分查找

```js
function binarySearch(array = [], target) {
  let l = 0;
  let r = array.length;
  // 在[l...r]都范围里寻找target
  while (l <= r) {
    // 当 l === r 时，区间[l...r]依然是有效的
    let mid = Math.ceil((l + r) / 2);
    if (array[mid] === target) {
      return mid;
    }
    if (target > array[mid]) {
      l = mid + 1; // target 在 [mid+1 ... r]中
    } else {
      r = mid - 1; // target 在 [l ... mid-1]中
    }
    return -1;
  }
}
```
